import { router } from '@kit.ArkUI';
import { ApiService } from '../services/ApiService';

interface PlanItem {
  id: number;
  name: string;
  year: number;
  province: string;
  score: number;
  subjectType: number;
  status: number;
  createTime: string;
}

interface DetailItem {
  id: number;
  planId: number;
  schoolId: number;
  schoolName: string;
  schoolCode: string;
  majorId: number;
  priority: number;
  probability: number;
  remark: string;
}

interface PlanListResponse {
  code: number;
  message: string;
  data: PlanItem[] | null;
}

interface PlanDetailData {
  plan: PlanItem;
  details: DetailItem[];
}

interface PlanDetailResponse {
  code: number;
  message: string;
  data: PlanDetailData | null;
}

@Entry
@Component
struct VolunteerPlan {
  @State plans: PlanItem[] = []
  @State currentPlan: PlanItem | null = null
  @State details: DetailItem[] = []
  @State loading: boolean = true
  @State fadeIn: boolean = false
  @State showCreate: boolean = false
  @State newPlanName: string = ''
  @State newPlanScore: string = ''
  @State newPlanProvince: string = ''

  aboutToAppear() {
    setTimeout(() => { this.fadeIn = true; }, 100);
    this.loadPlans();
  }

  async loadPlans() {
    this.loading = true;
    try {
      const httpRequest = await import('@ohos.net.http');
      const http = httpRequest.default.createHttp();
      const response = await http.request(
        `http://10.0.2.2:8080/api/volunteer/plans`,
        { method: httpRequest.default.RequestMethod.GET }
      );
      const parsed = JSON.parse(response.result as string) as PlanListResponse;
      http.destroy();

      if (parsed.code === 200 && parsed.data) {
        this.plans = parsed.data;
      }
    } catch (e) {
      console.error('鍔犺浇澶辫触');
    }
    this.loading = false;
  }

  async loadPlanDetail(planId: number) {
    try {
      const httpRequest = await import('@ohos.net.http');
      const http = httpRequest.default.createHttp();
      const response = await http.request(
        `http://10.0.2.2:8080/api/volunteer/plan/${planId}`,
        { method: httpRequest.default.RequestMethod.GET }
      );
      const result = JSON.parse(response.result as string) as PlanDetailResponse;
      http.destroy();

      if (result.code === 200 && result.data) {
        this.currentPlan = result.data.plan;
        this.details = result.data.details || [];
      }
    } catch (e) {
      console.error('鍔犺浇璇︽儏澶辫触');
    }
  }

  async createPlan() {
    if (!this.newPlanName || !this.newPlanScore) {
      return;
    }
    try {
      const httpRequest = await import('@ohos.net.http');
      const http = httpRequest.default.createHttp();
      await http.request(
        `http://10.0.2.2:8080/api/volunteer/create`,
        {
          method: httpRequest.default.RequestMethod.POST,
          header: { 'Content-Type': 'application/json' },
          extraData: JSON.stringify({
            name: this.newPlanName,
            score: parseInt(this.newPlanScore),
            year: 2024,
            province: this.newPlanProvince || '鏈煡',
            subjectType: 1
          })
        }
      );
      http.destroy();

      this.showCreate = false;
      this.newPlanName = '';
      this.newPlanScore = '';
      this.newPlanProvince = '';
      this.loadPlans();
    } catch (e) {
      console.error('鍒涘缓澶辫触');
    }
  }

  async deletePlan(planId: number) {
    try {
      const httpRequest = await import('@ohos.net.http');
      const http = httpRequest.default.createHttp();
      await http.request(
        `http://10.0.2.2:8080/api/volunteer/plan/${planId}`,
        { method: httpRequest.default.RequestMethod.DELETE }
      );
      http.destroy();

      this.currentPlan = null;
      this.details = [];
      this.loadPlans();
    } catch (e) {
      console.error('鍒犻櫎澶辫触');
    }
  }

  build() {
    Column() {
      // ==================== 椤堕儴瀵艰埅鏍?====================
      Row() {
        Text('鈥?)
          .fontSize(24)
          .fontColor('#182431')
          .fontFamily('HarmonyOS Sans')
          .fontWeight(FontWeight.Bold)
          .margin({ right: 16 })
          .onClick(() => {
            if (this.currentPlan) {
              this.currentPlan = null;
              this.details = [];
            } else {
              router.back();
            }
          })

        Text('蹇楁効鏂规')
          .fontSize(20)
          .fontWeight(FontWeight.Bold)
          .fontColor('#182431')
          .fontFamily('HarmonyOS Sans')
          .layoutWeight(1)

        if (!this.currentPlan) {
          Row() {
            Image($r('app.media.ic_add'))
              .width(20)
              .height(20)
              .margin({ right: 6 })
            Text('鏂板缓')
              .fontSize(14)
              .fontColor('#182431')
              .fontFamily('HarmonyOS Sans')
          }
          .padding({ left: 12, right: 12, top: 8, bottom: 8 })
          .backgroundColor('#F6F6F6')
          .borderRadius(16)
          .onClick(() => { this.showCreate = !this.showCreate; })
        }
      }
      .width('100%')
      .height(56)
      .padding({ left: 16, right: 16 })
      .backgroundColor('#FFFFFF')
      .border({ width: { bottom: 1 }, color: '#F0F0F0' })
      .shadow({ radius: 8, color: '#00000008', offsetY: 2 })

      // ==================== 鍐呭鍖哄煙 ====================
      if (this.currentPlan) {
        this.PlanDetailView()
      } else {
        this.PlanListView()
      }
    }
    .width('100%')
    .height('100%')
    .backgroundColor('#FFFFFF')
    .opacity(this.fadeIn ? 1 : 0)
    .animation({ duration: 300, curve: Curve.EaseOut })
  }

  @Builder
  PlanListView() {
    Column() {
      // 鍒涘缓琛ㄥ崟
      if (this.showCreate) {
        Column() {
          Text('鍒涘缓鏂版柟妗?)
            .fontSize(18)
            .fontWeight(FontWeight.Bold)
            .fontColor('#333333')
            .margin({ bottom: 16 })

          TextInput({ placeholder: '鏂规鍚嶇О', text: this.newPlanName })
            .width('100%')
            .height(48)
            .borderRadius(12)
            .backgroundColor('#F8F9FA')
            .border({ width: 1, color: '#E8E8E8' })
            .onChange((value: string) => { this.newPlanName = value; })
            .margin({ bottom: 12 })

          TextInput({ placeholder: '楂樿€冨垎鏁?, text: this.newPlanScore })
            .width('100%')
            .height(48)
            .borderRadius(12)
            .backgroundColor('#F8F9FA')
            .border({ width: 1, color: '#E8E8E8' })
            .onChange((value: string) => { this.newPlanScore = value; })
            .margin({ bottom: 12 })

          TextInput({ placeholder: '鐪佷唤锛堝彲閫夛級', text: this.newPlanProvince })
            .width('100%')
            .height(48)
            .borderRadius(12)
            .backgroundColor('#F8F9FA')
            .border({ width: 1, color: '#E8E8E8' })
            .onChange((value: string) => { this.newPlanProvince = value; })
            .margin({ bottom: 20 })

          Row() {
            Button('鍙栨秷')
              .width('45%')
              .height(48)
              .borderRadius(24)
              .backgroundColor('#F0F0F0')
              .fontColor('#666666')
              .onClick(() => { this.showCreate = false; })

            Button('鍒涘缓')
              .width('45%')
              .height(48)
              .borderRadius(24)
              .linearGradient({
                angle: 90,
                colors: [['#667eea', 0], ['#764ba2', 1]]
              })
              .fontColor('#FFFFFF')
              .onClick(() => { this.createPlan(); })
          }
          .width('100%')
          .justifyContent(FlexAlign.SpaceBetween)
        }
        .width('92%')
        .padding(24)
        .backgroundColor('#FFFFFF')
        .borderRadius(20)
        .margin({ top: 16, bottom: 16 })
        .shadow({ radius: 12, color: '#00000015', offsetY: 6 })
      }

      // 鏂规鍒楄〃
      if (this.loading) {
        Column() {
          Text('馃攧')
            .fontSize(48)
            .margin({ bottom: 12 })
          Text('鍔犺浇涓?..')
            .fontSize(16)
            .fontColor('#999999')
        }
        .layoutWeight(1)
        .justifyContent(FlexAlign.Center)
      } else if (this.plans.length === 0) {
        Column() {
          Column() {
            Text('馃搵')
              .fontSize(64)
              .margin({ bottom: 16 })
            Text('鏆傛棤蹇楁効鏂规')
              .fontSize(20)
              .fontWeight(FontWeight.Bold)
              .fontColor('#333333')
            Text('鐐瑰嚮涓婃柟鎸夐挳鍒涘缓')
              .fontSize(14)
              .fontColor('#999999')
              .margin({ top: 8 })
          }
          .padding(32)
          .backgroundColor('#FFFFFF')
          .borderRadius(24)
          .shadow({ radius: 12, color: '#00000012', offsetY: 6 })
        }
        .layoutWeight(1)
        .justifyContent(FlexAlign.Center)
      } else {
        List() {
          ForEach(this.plans, (plan: PlanItem) => {
            ListItem() {
              Row() {
                Column() {
                  Text(plan.name)
                    .fontSize(17)
                    .fontWeight(FontWeight.Medium)
                    .fontColor('#333333')

                  Row() {
                    Text(`${plan.year}骞碻)
                      .fontSize(13)
                      .fontColor('#999999')
                    if (plan.score) {
                      Text(` | ${plan.score}鍒哷)
                        .fontSize(13)
                        .fontColor('#667eea')
                        .fontWeight(FontWeight.Bold)
                    }
                  }
                  .margin({ top: 6 })
                }
                .layoutWeight(1)
                .alignItems(HorizontalAlign.Start)

                Text('鈥?)
                  .fontSize(24)
                  .fontColor('#CCCCCC')
              }
              .width('100%')
              .padding(16)
              .backgroundColor('#FFFFFF')
              .borderRadius(16)
              .margin({ bottom: 12 })
              .shadow({ radius: 6, color: '#00000012', offsetY: 3 })
              .onClick(() => { this.loadPlanDetail(plan.id); })
            }
          })
        }
        .width('92%')
        .layoutWeight(1)
        .margin({ top: 8 })
        .scrollBar(BarState.Off)
      }
    }
  }

  @Builder
  PlanDetailView() {
    Column() {
      // 鏂规淇℃伅
      if (this.currentPlan) {
        Column() {
          Row() {
            Text('馃搳')
              .fontSize(28)
              .margin({ right: 12 })

            Column() {
              Text(this.currentPlan.name)
                .fontSize(18)
                .fontWeight(FontWeight.Bold)
                .fontColor('#333333')

              Row() {
                Text(`${this.currentPlan.year}骞碻)
                  .fontSize(13)
                  .fontColor('#999999')
                if (this.currentPlan.score) {
                  Text(` | ${this.currentPlan.score}鍒哷)
                    .fontSize(13)
                    .fontColor('#667eea')
                    .fontWeight(FontWeight.Bold)
                }
              }
              .margin({ top: 4 })
            }
            .layoutWeight(1)
            .alignItems(HorizontalAlign.Start)
          }
          .width('100%')
        }
        .width('92%')
        .padding(20)
        .backgroundColor('#FFFFFF')
        .borderRadius(20)
        .margin({ top: 16, bottom: 16 })
        .shadow({ radius: 8, color: '#00000012', offsetY: 4 })
      }

      // 蹇楁効鍒楄〃
      if (this.details.length === 0) {
        Column() {
          Column() {
            Text('馃摑')
              .fontSize(64)
              .margin({ bottom: 16 })
            Text('鏆傛棤蹇楁効')
              .fontSize(20)
              .fontWeight(FontWeight.Bold)
              .fontColor('#333333')
            Text('浠庨櫌鏍¤鎯呴〉娣诲姞瀛︽牎鍒版柟妗?)
              .fontSize(14)
              .fontColor('#999999')
              .margin({ top: 8 })
          }
          .padding(32)
          .backgroundColor('#FFFFFF')
          .borderRadius(24)
          .shadow({ radius: 12, color: '#00000012', offsetY: 6 })
        }
        .layoutWeight(1)
        .justifyContent(FlexAlign.Center)
      } else {
        List() {
          ForEach(this.details, (item: DetailItem, index: number) => {
            ListItem() {
              Row() {
                // 鎺掑悕寰界珷
                Column() {
                  Text(`${index + 1}`)
                    .fontSize(16)
                    .fontWeight(FontWeight.Bold)
                    .fontColor('#FFFFFF')
                }
                .width(36)
                .height(36)
                .borderRadius(18)
                .linearGradient({
                  angle: 135,
                  colors: [['#667eea', 0], ['#764ba2', 1]]
                })
                .justifyContent(FlexAlign.Center)
                .margin({ right: 14 })

                Column() {
                  Text(item.schoolName || '鏈煡瀛︽牎')
                    .fontSize(16)
                    .fontWeight(FontWeight.Medium)
                    .fontColor('#333333')

                  if (item.probability) {
                    Row() {
                      Text('褰曞彇姒傜巼:')
                        .fontSize(12)
                        .fontColor('#999999')
                      Text(`${item.probability}%`)
                        .fontSize(12)
                        .fontColor('#4ECDC4')
                        .fontWeight(FontWeight.Bold)
                    }
                    .margin({ top: 4 })
                  }
                }
                .layoutWeight(1)
                .alignItems(HorizontalAlign.Start)
              }
              .width('100%')
              .padding(16)
              .backgroundColor('#FFFFFF')
              .borderRadius(16)
              .margin({ bottom: 12 })
              .shadow({ radius: 6, color: '#00000012', offsetY: 3 })
            }
          })
        }
        .width('92%')
        .layoutWeight(1)
        .scrollBar(BarState.Off)
      }

      // 鍒犻櫎鏂规鎸夐挳
      Button('鍒犻櫎姝ゆ柟妗?)
        .width('92%')
        .height(52)
        .borderRadius(26)
        .backgroundColor('#FF6B6B')
        .fontColor('#FFFFFF')
        .fontSize(16)
        .fontWeight(FontWeight.Bold)
        .shadow({ radius: 12, color: '#FF6B6B40', offsetY: 6 })
        .margin({ top: 16, bottom: 20 })
        .onClick(() => {
          if (this.currentPlan) {
            this.deletePlan(this.currentPlan.id);
          }
        })
    }
  }
}

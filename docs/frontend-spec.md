## 三、通用组件设计

### 3.1 导航栏

```typescript
@Builder
NavBar(title: string, showBack: boolean = true) {
    Row() {
        if (showBack) {
            Text('<').fontSize(24).onClick(() => router.back())
        }
        Text(title).fontSize(20).fontWeight(FontWeight.Bold)
    }
    .height(56)
    .backgroundColor('#1976D2')
    .fontColor('#FFFFFF')
}
```

### 3.2 学校卡片

```typescript
@Builder
SchoolCard(school: SchoolItem) {
    Row() {
        Column() {
            Text(school.name).fontSize(16).fontWeight(FontWeight.Medium)
            Row() {
                if (school.is985) Badge('985', '#FF6B6B')
                if (school.is211) Badge('211', '#4ECDC4')
                Text(school.typeName).fontSize(12).fontColor('#999')
            }
        }.layoutWeight(1)
        if (school.ranking) Text('#' + school.ranking).fontSize(16).fontColor('#1976D2')
    }
    .padding(12).backgroundColor('#FFF').borderRadius(8).margin({ bottom: 8 })
}
```

### 3.3 标签页

```typescript
@Builder
TabBar(tabs: string[], active: number, onChange: (i: number) => void) {
    Row() {
        ForEach(tabs, (tab: string, index: number) => {
            Text(tab)
                .fontSize(14)
                .fontColor(active === index ? '#1976D2' : '#666')
                .fontWeight(active === index ? FontWeight.Bold : FontWeight.Normal)
                .padding({ left: 16, right: 16, top: 8, bottom: 8 })
                .border({ width: { bottom: active === index ? 2 : 0 }, color: '#1976D2' })
                .onClick(() => onChange(index))
        })
    }
}
```
